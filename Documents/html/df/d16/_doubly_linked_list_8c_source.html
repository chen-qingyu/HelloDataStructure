107DoublyLinkedListItem DoublyLinkedList_At(const DoublyLinkedList* self, int index) // list[index] for index in range 0 to size will be O(1) on each access
108{
109 check_bounds(index, -self->size, self->size);
110
111 index = index >= 0 ? index : index + self->size;
112
113 DoublyLinkedList* list = (DoublyLinkedList*)self; // drop out const
114
115 // too far from the last accessed element
116 if (abs(index - self->latest) > self->size / 2)
117 {
118 // closer to the header or trailer
119 list->p_latest = (index < self->latest) ? self->header->next : self->trailer->prev;
120 list->latest = (index < self->latest) ? 0 : self->size - 1;
121 }
122
123 if (index < self->latest)
124 {
125 while (index < self->latest)
126 {
127 list->latest--;
128 list->p_latest = list->p_latest->prev;
129 }
130 }
131 else if (index > self->latest)
132 {
133 while (index > self->latest)
134 {
135 list->latest++;
136 list->p_latest = list->p_latest->next;
137 }
138 }
139 // else the element accessed this time is the same as the last time
140
141 return self->p_latest->data;
142}

144int DoublyLinkedList_Find(const DoublyLinkedList* self, DoublyLinkedListItem data)
145{
146 int index = 0;
147 struct DoublyLinkedListNode* current = self->header->next;
148
149 while (current != self->trailer && current->data != data)
150 {
151 current = current->next;
152 index++;
153 }
154
155 return current != self->trailer ? index : -1;

237

239 current->prev->next = current->next;
240 current->next->prev = current->prev;
241 free(current);
242
243 // resize
244 --self->size;
