102// Traverse the tree in specified order.
103static void traverse(const struct BinarySearchTreeNode* node, enum TraverseOption order, void (*p_trav)(const BinarySearchTreeItem data))
104{
105 if (node)
106 {
107 switch (order)
108 {
109 case PRE_ORDER:
110 {
111 p_trav(node->data);
112 traverse(node->left, PRE_ORDER, p_trav);
113 traverse(node->right, PRE_ORDER, p_trav);
114 break;
115 }
116
117 case IN_ORDER:
118 {
119 traverse(node->left, IN_ORDER, p_trav);
120 p_trav(node->data);
121 traverse(node->right, IN_ORDER, p_trav);
122 break;
123 }
124
125 case POST_ORDER:
126 {
127 traverse(node->left, POST_ORDER, p_trav);
128 traverse(node->right, POST_ORDER, p_trav);
129 p_trav(node->data);
130 break;
131 }
132
133 case LEVEL_ORDER:
134 {
135 ArrayQueue* queue = ArrayQueue_Create();
136 ArrayQueue_Enqueue(queue, node);
137 while (!ArrayQueue_IsEmpty(queue))
138 {
139 node = ArrayQueue_Dequeue(queue);
140 p_trav(node->data);
141 if (node->left)
142 {
143 ArrayQueue_Enqueue(queue, node->left);
144 }
145 if (node->right)
146 {
147 ArrayQueue_Enqueue(queue, node->right);
148 }
149 }
150 ArrayQueue_Destroy(queue);
151 break;
152 }
153
154 default:
155 {
156 fprintf(stderr, "ERROR: Invalid order for traverse.\n");
157 exit(EXIT_FAILURE);
158 break;
159 }
160 }
161 }
162}

166{

203// Remove node recursively.
204static struct BinarySearchTreeNode* remove_node(BinarySearchTree* tree, struct BinarySearchTreeNode* node, BinarySearchTreeItem data)
205{
206 if (node)
207 {
208 if (data < node->data)
209 {
210 node->left = remove_node(tree, node->left, data);
211 }
212 else if (data > node->data)
213 {
214 node->right = remove_node(tree, node->right, data);
215 }
216 else // data == node->data
217 {
218 if (node->left && node->right)
219 {
220 struct BinarySearchTreeNode* tmp = find_min(node->right); // node->right is not NULL
221 node->data = tmp->data;
222 node->right = remove_node(tree, node->right, tmp->data);
223 }
224 else
225 {
226 struct BinarySearchTreeNode* tmp = node;
227 node = node->left ? node->left : node->right;
228 free(tmp);
229 tree->size--;
230 }
231 }
232 }
233
234 return node;
235}
