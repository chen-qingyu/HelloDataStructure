47static int hash(HashTableKey key, int mod)
48{
49 unsigned int index = 0;
50
51 while (*key != '\0')
52 {
53 index = (index << 5) + *key++;
54 }
55
56 return index % mod;
57}

59static int find_pos(const HashTable* table, HashTableKey key)
60{
61 int current_pos = hash(key, table->capacity);
62 int new_pos = current_pos;
63 int conflict_cnt = 0;
64
65 while (table->pairs[new_pos].state != EMPTY && strcmp(table->pairs[new_pos].key, key) != 0)
66 {
67 if (++conflict_cnt % 2)
68 {
69 new_pos = current_pos + (conflict_cnt + 1) * (conflict_cnt + 1) / 4;
70 if (new_pos >= table->capacity)
71 {
72 new_pos %= table->capacity;
73 }
74 }
75 else
76 {
77 new_pos = current_pos - conflict_cnt * conflict_cnt / 4;
78 while (new_pos < 0)
79 {
80 new_pos += table->capacity;
81 }
82 }
83 }
84
85 return new_pos;
86}
